1. Logística de Reparto

Inicio
    // Cargar ubicaciones de los puntos de entrega y el almacén
    Definir ubicaciones Como Lista de Puntos
    Definir ruta Como Lista de Puntos
    Definir distanciaTotal Como Entero

    // Seleccionar el almacén como punto inicial
    puntoActual <- ubicaciones[0]
    ruta <- [puntoActual]
    distanciaTotal <- 0

    // Construir la ruta usando el vecino más cercano
    Mientras longitud(ruta) < longitud(ubicaciones) Hacer
        vecinoMasCercano <- EncontrarVecinoMasCercano(puntoActual, ubicaciones, ruta)
        distanciaTotal <- distanciaTotal + Distancia(puntoActual, vecinoMasCercano)
        puntoActual <- vecinoMasCercano
        ruta <- ruta + [puntoActual]
    FinMientras

    // Regresar al almacén
    distanciaTotal <- distanciaTotal + Distancia(puntoActual, ubicaciones[0])
    ruta <- ruta + [ubicaciones[0]]

    // Mostrar la ruta y la distancia total
    Escribir "Ruta óptima: ", ruta
    Escribir "Distancia total: ", distanciaTotal
Fin
