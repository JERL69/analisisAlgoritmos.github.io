Inicio
    // Definir los puntos y sus coordenadas
    Puntos = [(x1, y1), (x2, y2), ..., (xn, yn)]
    Inicio = Puntos[0]  // Elegir un punto de inicio (puede ser cualquiera)

    // Inicializar variables
    Recorrido = []  // Almacenará el orden de los puntos visitados
    Visitados = []   // Almacenará los puntos ya visitados
    PuntoActual = Inicio
    Visitados.append(PuntoActual)
    Recorrido.append(PuntoActual)

    // Construir el recorrido
    Mientras longitud(Visitados) < longitud(Puntos):
        // Encontrar el punto más cercano no visitado
        PuntoCercano = EncontrarPuntoMasCercano(PuntoActual, Puntos, Visitados)
        
        // Agregar el punto más cercano al recorrido
        Recorrido.append(PuntoCercano)
        Visitados.append(PuntoCercano)
        PuntoActual = PuntoCercano

    // Regresar al punto de inicio
    Recorrido.append(Inicio)

    // Calcular la distancia total del recorrido
    DistanciaTotal = CalcularDistanciaTotal(Recorrido)

    // Mostrar resultados
    Mostrar "Recorrido óptimo: ", Recorrido
    Mostrar "Distancia total: ", DistanciaTotal
Fin

// Función para encontrar el punto más cercano no visitado
Función EncontrarPuntoMasCercano(PuntoActual, Puntos, Visitados):
    PuntoCercano = null
    DistanciaMinima = infinito

    Para cada Punto en Puntos:
        Si Punto no está en Visitados:
            Distancia = CalcularDistancia(PuntoActual, Punto)
            Si Distancia < DistanciaMinima:
                DistanciaMinima = Distancia
                PuntoCercano = Punto

    Retornar PuntoCercano

// Función para calcular la distancia entre dos puntos
Función CalcularDistancia(Punto1, Punto2):
    dx = Punto2.x - Punto1.x
    dy = Punto2.y - Punto1.y
    Retornar sqrt(dx^2 + dy^2)

// Función para calcular la distancia total del recorrido
Función CalcularDistanciaTotal(Recorrido):
    DistanciaTotal = 0
    Para i desde 0 hasta longitud(Recorrido) - 2:
        DistanciaTotal += CalcularDistancia(Recorrido[i], Recorrido[i+1])
    Retornar DistanciaTotal
