Diagrama de Flujo - Movimiento del Caballo

Algoritmo de Búsqueda del Movimiento del Caballo

        graph TD;
            A[Inicio] --> B[Definir tablero y movimientos válidos];
            B --> C[Establecer posición inicial del caballo];
            C --> D[Marcar casilla inicial como visitada];
            D --> E[Intentar mover el caballo a una nueva casilla];
            E -->|¿Movimiento válido?| F{Sí};
            E -->|No| G[Retroceder al último movimiento];
            G --> E;
            F --> H[Marcar casilla como visitada];
            H --> I[¿Se han visitado todas las casillas?];
            I -->|Sí| J[Mostrar solución];
            I -->|No| E;
            J --> K[Fin];
